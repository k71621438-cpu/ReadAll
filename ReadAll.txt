fun verifyAndUnpack(response: JSONObject, sentRid: String): JSONObject? {
    val p     = response.getJSONObject("p")
    val t     = response.getString("t")
    val n     = response.getString("n")
    val s     = response.getString("s")
    val ecRid = response.getString("rid")

    if (ecRid != sentRid) return null          // replay / wrong response

    val ageSeconds = abs(System.currentTimeMillis()/1000 - parseIso(t))
    if (ageSeconds > 300) return null          // too old

    val pHash     = sha256(jsonCanonical(p))
    val canonical = "$pHash.$t.$sentRid.$n"
    val expected  = hmacSha256(HMAC_SECRET, canonical)
    if (!expected.equals(s, ignoreCase = true)) return null   // tampered

    return p   // safe to read now
}

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
FaceGate Activation Server — Hardened Edition
==============================================

SECURITY MODEL
--------------
1. HMAC-signed responses — every response is wrapped in a signed envelope.
   A MITM cannot flip ok/access fields without knowing RESPONSE_HMAC_SECRET,
   which is baked into the APK (not transmitted over the wire).

2. Trial re-use prevention — trial is bound to THREE independent fingerprints:
     • device_id   — the app-generated ID (changes if app data is cleared)
     • android_id  — Android Settings.Secure.ANDROID_ID (survives reinstall,
                     persists until factory reset)
     • wifi_bssid  — MAC address of the connected router (not the IP, which
                     changes). Blocks a different device on the same WiFi.
   A trial is blocked if ANY of those three matches a previous trial.

3. Server-side trial timer — the server owns started_at / expires_at.
   App calls /api/heartbeat every 60 s; server replies with
   remaining_seconds (authoritative) and destruct: true when time is up.
   App never trusts its own local clock for trial state.

4. Paid licences — permanent per device. Once device_id + token_hash are
   stored in device_activations the device has access forever, regardless
   of whether the key is later exhausted on other devices.
   The only revocation is clearing app data (changes device_id).

5. Replay protection — every request carries a client-generated rid (UUID).
   The server echoes rid inside the signed payload. The app rejects any
   response where rid ≠ what it sent, or where timestamp is >5 minutes old.

--------------------
• Set FACEGATE_HMAC_SECRET env var to a long random string.
  Bake the SAME value into the APK as a compile-time constant.
• Set FACEGATE_ADMIN_KEY env var.
• Run behind gunicorn/uwsgi, not Flask dev server.

def _now() -> datetime:
    return datetime.now(timezone.utc)

def _now_iso() -> str:
    return _now().isoformat()

def _parse_iso(s: str) -> Optional[datetime]:
    if not s:
        return None
    try:
        dt = datetime.fromisoformat(s)
        if dt.tzinfo is None:
            dt = dt.replace(tzinfo=timezone.utc)
        return dt
    except Exception:
        return None


# ─────────────────────────────────────────────────────────────────────────────
# CRYPTO HELPERS
# ─────────────────────────────────────────────────────────────────────────────

def _hmac_hex(key: str, message: str) -> str:
    return hmac.new(
        key.encode("utf-8"),
        message.encode("utf-8"),
        hashlib.sha256
    ).hexdigest()


def _derive_token(device_id: str, key_id: int, key_text: str) -> str:
    """
    Deterministically derive a token for (device, activation_key).
    The same device re-activating the same key always gets the same token.
    This is intentional — it lets the app re-activate after a reinstall
    on the same device_id without consuming another slot.
    """
    raw = f"facegate:paid:{device_id}:{key_id}:{key_text}"
    return _hmac_hex(RESPONSE_HMAC_SECRET, raw)


def _hash_token(token: str) -> str:
    """One-way hash stored in DB so the raw token is never at rest."""
    return hashlib.sha256(token.encode()).hexdigest()


def _sign_response(payload: dict, rid: str) -> dict:
    """
    Wrap payload in a signed envelope.

    Envelope structure:
        {
            "p":   { ...payload... },
            "t":   "2024-...",        # server UTC timestamp ISO-8601
            "rid": "<echoed rid>",    # must match what client sent
            "n":   "<16-byte hex>",   # random nonce
            "s":   "<sha256 hmac>"    # signature (see below)
        }

    Signature input:
        SHA256( canonical_json(p) ) + "." + t + "." + rid + "." + n

    App verification steps (MUST ALL PASS before trusting p):
        1. Assert response["rid"] == rid you sent in request
        2. Assert abs(now - parse(response["t"])) < 300 seconds
        3. Recompute: ph = SHA256(json(p, sorted, no-spaces))
           canonical  = ph + "." + t + "." + rid + "." + n
           expected_s = HMAC-SHA256(RESPONSE_HMAC_SECRET, canonical)
        4. Assert response["s"] == expected_s  (constant-time compare)
        5. Only now read response["p"]
    """
    ts    = _now_iso()
    nonce = secrets.token_hex(16)

    payload_bytes = json.dumps(payload, separators=(",", ":"), sort_keys=True).encode()
    payload_hash  = hashlib.sha256(payload_bytes).hexdigest()
    canonical     = f"{payload_hash}.{ts}.{rid}.{nonce}"
    sig           = _hmac_hex(RESPONSE_HMAC_SECRET, canonical)

    return {"p": payload, "t": ts, "rid": rid, "n": nonce, "s": sig}

def _fp(data: dict) -> dict:
    """Extract all device fingerprint fields from a request body."""
    return {
        "device_id":  (data.get("device_id")  or "").strip(),
        "android_id": (data.get("android_id") or "").strip(),
        # BSSID = MAC address of the WiFi router — much harder to spoof than IP
        "wifi_bssid": (data.get("wifi_bssid") or "").strip().upper(),
        "wifi_ip":    (data.get("wifi_ip")    or "").strip(),
        "build_fp":   (data.get("build_fp")   or "").strip(),
    }


def _err(msg: str, rid: str, code: int = 400):
    return jsonify(_sign_response({"ok": False, "reason": msg}, rid)), code

def _trial_fingerprint_used(c: sqlite3.Cursor, fp: dict) -> bool:
    """
    Returns True if ANY fingerprint matches a previous trial — active or expired.
    Rows are never deleted so history is permanent.

    Checks independently:
      • device_id   — same install
      • android_id  — same physical device (survives app-data clear)
      • wifi_bssid  — same router (blocks a friend on same network)
    """
    if fp["device_id"]:
        c.execute("SELECT 1 FROM trial_usage WHERE device_id = ? LIMIT 1",
                  (fp["device_id"],))
        if c.fetchone():
            return True

    if fp["android_id"]:
        c.execute("SELECT 1 FROM trial_usage WHERE android_id = ? LIMIT 1",
                  (fp["android_id"],))
        if c.fetchone():
            return True

    if fp["wifi_bssid"]:
        c.execute("SELECT 1 FROM trial_usage WHERE wifi_bssid = ? LIMIT 1",
                  (fp["wifi_bssid"],))
        if c.fetchone():
            return True

    return False


def _active_trial_row(c: sqlite3.Cursor, device_id: str, android_id: str = ""):
    """Return the most recent active trial row for this device, or None."""
    c.execute("""
        SELECT * FROM trial_usage
        WHERE device_id = ? AND is_active = 1
        ORDER BY id DESC LIMIT 1
    """, (device_id,))
    row = c.fetchone()
    if row:
        return row

    # Fallback: look up by android_id in case device_id changed (reinstall)
    if android_id:
        c.execute("""
            SELECT * FROM trial_usage
            WHERE android_id = ? AND is_active = 1
            ORDER BY id DESC LIMIT 1
        """, (android_id,))
        row = c.fetchone()

    return row

@app.route("/api/activate", methods=["POST"])
def activate():
    """
    Activate a paid key or start a trial.

    Request body (JSON):
        key        : "NOWORNEVER" for trial, or a paid activation key
        device_id  : app-generated unique device identifier
        android_id : android.provider.Settings.Secure.ANDROID_ID
        wifi_bssid : WifiInfo.getBSSID()  — router MAC e.g. "AA:BB:CC:DD:EE:FF"
        wifi_ip    : WifiInfo.getIpAddress() as string (optional but helpful)
        build_fp   : android.os.Build.FINGERPRINT (optional)
        rid        : client-generated UUID for this request

    Successful paid response payload (inside "p"):
        { ok: true, access: "paid", token: "<hex>", expires_at: null }

    Successful trial response payload (inside "p"):
        { ok: true, access: "trial", started_at: "...", expires_at: "...",
          remaining_seconds: 3600 }

    Error response payload (inside "p"):
        { ok: false, reason: "<code>" }
        reason codes: invalid_key | key_exhausted | trial_exhausted | server_error
    """
    data = request.get_json(silent=True) or {}
    rid  = (data.get("rid") or secrets.token_hex(8))

    if not data.get("key") or not data.get("device_id"):
        return _err("missing required fields: key, device_id", rid)

    fingerprint = _fp(data)
    key         = data["key"].strip().upper()

    if key == "NOWORNEVER":
        return _activate_trial(fingerprint, rid)
    return _activate_paid(fingerprint, key, rid)


def _activate_trial(fp: dict, rid: str):
    conn = get_db()
    c    = conn.cursor()
    try:
        # ── Already has an active trial? Re-issue timing info. ──────────────
        existing = _active_trial_row(c, fp["device_id"], fp["android_id"])
        if existing:
            now        = _now()
            expires_at = _parse_iso(existing["expires_at"])
            remaining  = max(0, int((expires_at - now).total_seconds())) if expires_at else 0

            if remaining <= 0:
                # Trial expired — mark inactive and reject
                c.execute("UPDATE trial_usage SET is_active = 0 WHERE id = ?",
                          (existing["id"],))
                conn.commit()
                return _err("trial_expired", rid, 403)

            payload = {
                "ok":               True,
                "access":           "trial",
                "started_at":       existing["started_at"],
                "expires_at":       existing["expires_at"],
                "remaining_seconds": remaining,
            }
            return jsonify(_sign_response(payload, rid)), 200

        # ── Check all fingerprints against historical trial records ─────────
        if _trial_fingerprint_used(c, fp):
            return _err("trial_exhausted", rid, 403)

        # ── Grant new trial ─────────────────────────────────────────────────
        now        = _now()
        expires_at = now + timedelta(hours=TRIAL_HOURS)
        now_iso    = now.isoformat()
        exp_iso    = expires_at.isoformat()

        c.execute("""
            INSERT INTO trial_usage
              (device_id, android_id, wifi_bssid, wifi_ip, build_fp,
               started_at, expires_at, is_active, last_heartbeat)
            VALUES (?, ?, ?, ?, ?, ?, ?, 1, ?)
        """, (fp["device_id"], fp["android_id"], fp["wifi_bssid"],
              fp["wifi_ip"],   fp["build_fp"],   now_iso, exp_iso, now_iso))
        conn.commit()

        payload = {
            "ok":               True,
            "access":           "trial",
            "started_at":       now_iso,
            "expires_at":       exp_iso,
            "remaining_seconds": int(TRIAL_HOURS * 3600),
        }
        return jsonify(_sign_response(payload, rid)), 200

    except Exception as e:
        conn.rollback()
        print(f"❌ trial activation error: {e}")
        return _err("server_error", rid, 500)
    finally:
        conn.close()


def _activate_paid(fp: dict, key: str, rid: str):
    conn = get_db()
    c    = conn.cursor()
    try:
        # ── Lookup key ──────────────────────────────────────────────────────
        c.execute("SELECT * FROM activation_keys WHERE key = ?", (key,))
        key_rec = c.fetchone()
        if not key_rec:
            return _err("invalid_key", rid, 403)

        # ── Already activated on this device? Re-issue token. ───────────────
        c.execute("""
            SELECT * FROM device_activations
            WHERE key_id = ? AND device_id = ?
        """, (key_rec["id"], fp["device_id"]))
        existing = c.fetchone()

        if existing:
            token = _derive_token(fp["device_id"], key_rec["id"], key_rec["key"])
            c.execute("UPDATE device_activations SET last_seen = ? WHERE id = ?",
                      (_now_iso(), existing["id"]))
            conn.commit()
            payload = {"ok": True, "access": "paid", "token": token, "expires_at": None}
            return jsonify(_sign_response(payload, rid)), 200

        # ── Check remaining slots ───────────────────────────────────────────
        if key_rec["used_count"] >= key_rec["max_devices"]:
            return _err("key_exhausted", rid, 403)

        # ── Activate ────────────────────────────────────────────────────────
        token      = _derive_token(fp["device_id"], key_rec["id"], key_rec["key"])
        token_hash = _hash_token(token)
        now        = _now_iso()
        new_count  = key_rec["used_count"] + 1
        new_status = "exhausted" if new_count >= key_rec["max_devices"] else "partial"

        c.execute("""
            INSERT INTO device_activations
              (key_id, device_id, android_id, wifi_bssid, wifi_ip, build_fp,
               token_hash, activated_at, last_seen)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (key_rec["id"], fp["device_id"], fp["android_id"],
              fp["wifi_bssid"], fp["wifi_ip"], fp["build_fp"],
              token_hash, now, now))

        c.execute("""
            UPDATE activation_keys
            SET used_count = ?, status = ?, updated_at = ?
            WHERE id = ?
        """, (new_count, new_status, now, key_rec["id"]))

        conn.commit()

        payload = {"ok": True, "access": "paid", "token": token, "expires_at": None}
        return jsonify(_sign_response(payload, rid)), 200

    except Exception as e:
        conn.rollback()
        print(f"❌ paid activation error: {e}")
        return _err("server_error", rid, 500)
    finally:
        conn.close()

@app.route("/api/verify", methods=["POST"])
def verify():
    """
    Periodic access check. Call every ~5 minutes from the app.

    For paid devices:
        Body: { device_id, token, rid }

    For trial devices:
        Body: { device_id, android_id, rid }
        (no token — trial is looked up by device fingerprint)

    Response payload:
        Paid:  { ok: true,  access: "paid",  token, expires_at: null }
        Trial: { ok: true,  access: "trial", remaining_seconds, expires_at, destruct: false }
        Expired/invalid: { ok: false, access: "none", reason, destruct: true }

    The app must:
      1. Verify the HMAC signature.
      2. If destruct == true, immediately stop all camera hooks and show
         an expiry/invalid screen. Do not wait for the next heartbeat.
    """
    data = request.get_json(silent=True) or {}
    rid  = (data.get("rid") or secrets.token_hex(8))

    device_id  = (data.get("device_id")  or "").strip()
    android_id = (data.get("android_id") or "").strip()
    token      = (data.get("token")      or "").strip()

    if not device_id:
        return _err("missing device_id", rid)

    conn = get_db()
    c    = conn.cursor()
    try:
        # ── Paid path ────────────────────────────────────────────────────────
        if token:
            token_hash = _hash_token(token)
            c.execute("""
                SELECT da.id, da.last_seen, ak.key
                FROM device_activations da
                JOIN activation_keys ak ON ak.id = da.key_id
                WHERE da.device_id = ? AND da.token_hash = ?
            """, (device_id, token_hash))
            row = c.fetchone()

            if not row:
                payload = {"ok": False, "access": "none",
                           "reason": "not_activated", "destruct": True}
                return jsonify(_sign_response(payload, rid)), 403

            c.execute("UPDATE device_activations SET last_seen = ? WHERE id = ?",
                      (_now_iso(), row["id"]))
            conn.commit()

            payload = {"ok": True, "access": "paid", "token": token,
                       "expires_at": None, "destruct": False}
            return jsonify(_sign_response(payload, rid)), 200

        # ── Trial path ───────────────────────────────────────────────────────
        row = _active_trial_row(c, device_id, android_id)

        if not row:
            payload = {"ok": False, "access": "none",
                       "reason": "no_trial", "destruct": True}
            return jsonify(_sign_response(payload, rid)), 403

        now        = _now()
        expires_at = _parse_iso(row["expires_at"])
        remaining  = int((expires_at - now).total_seconds()) if expires_at else 0

        c.execute("UPDATE trial_usage SET last_heartbeat = ? WHERE id = ?",
                  (now.isoformat(), row["id"]))

        if remaining <= 0:
            c.execute("UPDATE trial_usage SET is_active = 0 WHERE id = ?", (row["id"],))
            conn.commit()
            payload = {"ok": False, "access": "none", "reason": "trial_expired",
                       "remaining_seconds": 0, "expires_at": row["expires_at"],
                       "destruct": True}
            return jsonify(_sign_response(payload, rid)), 200

        conn.commit()
        payload = {"ok": True, "access": "trial", "remaining_seconds": remaining,
                   "expires_at": row["expires_at"], "destruct": False}
        return jsonify(_sign_response(payload, rid)), 200

    except Exception as e:
        print(f"❌ verify error: {e}")
        return _err("server_error", rid, 500)
    finally:
        conn.close()

@app.route("/api/heartbeat", methods=["POST"])
def heartbeat():
    """
    Lightweight timer ping. Trial apps call this every 60 seconds.
    Paid apps do NOT need to call this — use /api/verify on a longer interval.

    Request body: { device_id, android_id, rid }

    Response payload:
        Active:  { ok: true,  destruct: false, remaining_seconds: N }
        Expired: { ok: false, destruct: true,  remaining_seconds: 0, reason: "trial_expired" }
        Unknown: { ok: false, destruct: true,  reason: "no_trial" }

    When the app receives destruct: true it MUST:
      • Stop injecting frames immediately
      • Close camera session
      • Show "Trial expired" UI
      • Not retry heartbeat until user re-activates
    """
    data = request.get_json(silent=True) or {}
    rid  = (data.get("rid") or secrets.token_hex(8))

    device_id  = (data.get("device_id")  or "").strip()
    android_id = (data.get("android_id") or "").strip()

    if not device_id:
        return _err("missing device_id", rid)

    conn = get_db()
    c    = conn.cursor()
    try:
        row = _active_trial_row(c, device_id, android_id)

        if not row:
            payload = {"ok": False, "destruct": True, "reason": "no_trial"}
            return jsonify(_sign_response(payload, rid)), 200

        now        = _now()
        expires_at = _parse_iso(row["expires_at"])
        remaining  = int((expires_at - now).total_seconds()) if expires_at else 0

        c.execute("UPDATE trial_usage SET last_heartbeat = ? WHERE id = ?",
                  (now.isoformat(), row["id"]))

        if remaining <= 0:
            c.execute("UPDATE trial_usage SET is_active = 0 WHERE id = ?", (row["id"],))
            conn.commit()
            payload = {"ok": False, "destruct": True,
                       "reason": "trial_expired", "remaining_seconds": 0}
            return jsonify(_sign_response(payload, rid)), 200

        conn.commit()
        payload = {"ok": True, "destruct": False, "remaining_seconds": remaining}
        return jsonify(_sign_response(payload, rid)), 200

    except Exception as e:
        print(f"❌ heartbeat error: {e}")
        return _err("server_error", rid, 500)
    finally:
        conn.close()

@app.route("/api/admin/revoke_device", methods=["POST"])
def admin_revoke_device():
    """
    Revoke a specific device activation (e.g. fraudulent use).
    Body: { key, device_id }
    Headers: X-Admin-Key: <FACEGATE_ADMIN_KEY>

    This deletes the device_activation row AND decrements used_count so the
    slot can be re-used. The device's token becomes invalid immediately.
    """
    _require_admin()
    data      = request.get_json(silent=True) or {}
    key       = (data.get("key") or "").strip().upper()
    device_id = (data.get("device_id") or "").strip()

    if not key or not device_id:
        return jsonify({"error": "missing key or device_id"}), 400

    conn = get_db()
    c    = conn.cursor()
    try:
        c.execute("SELECT * FROM activation_keys WHERE key = ?", (key,))
        rec = c.fetchone()
        if not rec:
            return jsonify({"error": "key not found"}), 404

        c.execute("""
            DELETE FROM device_activations WHERE key_id = ? AND device_id = ?
        """, (rec["id"], device_id))
        deleted = c.rowcount

        if deleted:
            new_count  = max(0, rec["used_count"] - 1)
            new_status = "unused" if new_count == 0 else "partial"
            c.execute("""
                UPDATE activation_keys
                SET used_count = ?, status = ?, updated_at = ?
                WHERE id = ?
            """, (new_count, new_status, _now_iso(), rec["id"]))

        conn.commit()
        return jsonify({"revoked": bool(deleted), "device_id": device_id})
    finally:
        conn.close()
